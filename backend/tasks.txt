    async saveService(userId: string, insuranceId: string) {
        const user = await User.findById(userId);
        if(!user.savedInsurancePlans.includes(insuranceId)){
            user.savedInsurancePlans.push(insuranceId);
            await user.save();
        }
        return user;
    }

    @Post(':userId/save-insurance/:insuranceId')
    async saveInsurance(@Param('userId') userId: string, @Param('insuranceId') insuranceId: string){
        return this.userService.saveService(userId, insuranceId);
    }
